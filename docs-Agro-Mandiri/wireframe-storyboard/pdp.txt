        // Change main product image when thumbnail is clicked
        function changeImage(src) {
            document.getElementById('mainImage').src = src;
            
            // Update active state for thumbnails
            const thumbnails = document.querySelectorAll('.image-thumbnail');
            thumbnails.forEach(thumb => {
                thumb.classList.remove('active');
                if (thumb.querySelector('img').src.includes(src.split('?')[1])) {
                    thumb.classList.add('active');
                }
            });
        }

        // Select product variant
        function selectVariant(variant) {
            const variantButtons = document.querySelectorAll('.variant-btn');
            variantButtons.forEach(btn => {
                btn.classList.remove('active');
                if (btn.textContent.trim().includes(variant)) {
                    btn.classList.add('active');
                }
            });
            
            // Here you would typically update pricing and other details based on variant
            // For demo purposes, we'll just log the selected variant
            console.log('Selected variant:', variant);
        }

        // Quantity controls
        function incrementQuantity() {
            const quantityInput = document.getElementById('quantity');
            quantityInput.value = parseInt(quantityInput.value) + 1;
            updatePrice();
        }

        function decrementQuantity() {
            const quantityInput = document.getElementById('quantity');
            if (parseInt(quantityInput.value) > 1) {
                quantityInput.value = parseInt(quantityInput.value) - 1;
                updatePrice();
            }
        }

        // Select pricing tier
        function selectTier(min, max, pricePerKg) {
            const tiers = document.querySelectorAll('.price-tier');
            tiers.forEach(tier => {
                tier.classList.remove('active');
                
                // Show tier total for all tiers (for demo purposes)
                const tierTotal = tier.querySelector('.tier-total');
                tierTotal.classList.remove('hidden');
            });
            
            // Highlight selected tier
            event.currentTarget.classList.add('active');
            
            // Update quantity input range
            const quantityInput = document.getElementById('quantity');
            quantityInput.min = min;
            quantityInput.max = max;
            
            // Store the selected price per kg for calculations
            quantityInput.dataset.pricePerKg = pricePerKg;
            
            // Update price display
            updatePrice();
        }

        // Update price based on quantity and selected tier
        function updatePrice() {
            const quantityInput = document.getElementById('quantity');
            const quantity = parseInt(quantityInput.value);
            const pricePerKg = parseFloat(quantityInput.dataset.pricePerKg || 10000); // Default to first tier
            
            const totalPrice = quantity * 50 * pricePerKg; // 50 kg per sack
            const retailPrice = quantity * 50 * 11000; // Retail price is 11,000/kg
            const savings = retailPrice - totalPrice;
            
            // Update price summary
            document.querySelector('.price-summary .subtotal').textContent = `Rp. ${totalPrice.toLocaleString('id-ID')}`;
            document.querySelector('.price-summary .savings').textContent = `Rp. ${savings.toLocaleString('id-ID')}`;
            document.querySelector('.price-summary .total').textContent = `Rp. ${totalPrice.toLocaleString('id-ID')}`;
        }

        // Tab functionality
        function openTab(evt, tabName) {
            const tabContents = document.querySelectorAll('.tab-content');
            tabContents.forEach(content => {
                content.classList.remove('active');
            });
            
            const tabButtons = document.querySelectorAll('.tab-button');
            tabButtons.forEach(button => {
                button.classList.remove('border-green-500', 'text-green-600');
                button.classList.add('border-transparent', 'text-gray-500');
            });
            
            document.getElementById(tabName).classList.add('active');
            evt.currentTarget.classList.add('border-green-500', 'text-green-600');
            evt.currentTarget.classList.remove('border-transparent', 'text-gray-500');
        }

        // Initialize with first tier selected
        document.addEventListener('DOMContentLoaded', function() {
            selectTier(1, 39, 10000);
            
            // Make header sticky on scroll
            const header = document.querySelector('.sticky-header');
            window.addEventListener('scroll', function() {
                if (window.scrollY > 50) {
                    header.classList.add('scrolled');
                } else {
                    header.classList.remove('scrolled');
                }
            });
            
            // Update price when quantity input changes
            document.getElementById('quantity').addEventListener('change', updatePrice);
        });
